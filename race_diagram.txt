@startuml
entity System {
  Id: Guid <<PK>>
  Name: varchar(150)
}

entity Race {
  Id: Guid <<PK>>
  ParentRaceId: Guid null <<FK>>
  SystemId: Guid <<FK>>
  Name: varchar(150)
  Description: varchar(150)
  IsSubRace: bit
  Speed: float
}

entity Language {
  Id: Guid <<PK>>
  RaceId: Guid <<FK>>
  Name: varchar(30)
  Alphabet: varchar(20)
}

entity RaceLanguage {
  RaceId: Guid <<PK>>
  LanguageId: Guid <<PK>>
}

entity AbilityScore {
  Id: Guid <<PK>>
  Name: varchar(30)
  Abbreviation: varchar(3)
  Description: text
}

entity RaceAbilityScore {
  RaceId: Guid <<PK>> <<FK>>
  AbilityScoreId: Guid <<PK>> <<FK>>
  AddingValue: int
}

entity Feature {
  Id: Guid <<PK>>
  Name: varchar(50)
}

entity RaceFeature {
  RaceId: Guid <<PK>> <<FK>>
  Feature: Guid <<PK>> <<FK>>
}

entity Alignment {
  Id: Guid <<PK>>
  Name: varchar(15)
}

entity RaceAlignment {
  RaceId: Guid <<PK>> <<FK>>
  AlignmentId: Guid <<PK>> <<FK>>
}

Race -- RaceLanguage
Language -- RaceLanguage

Race -- RaceAbilityScore
AbilityScore -- RaceAbilityScore

Race -- RaceFeature
Feature -- RaceFeature

Race -- RaceAlignment
Alignment -- RaceAlignment
@enduml
